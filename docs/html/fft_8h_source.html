33  {
34  static_assert(std::is_floating_point_v<FloatT>);
35  namespace numbers = devfix::base::numbers;
36  if (math::exp2(math::floorLog2(len)) != len) { throw std::invalid_argument("len is not a power of 2"); }
37 
38  const auto theta = numbers::pi_v<FloatT> / len;
39  std::complex<FloatT> phi = std::complex<FloatT>(cos(theta), -sin(theta)), w;
40  for (std::size_t stage = len; stage > 1; stage >>= 1u)
41  {
42  phi *= phi;
43  w = 1;
44  for (std::size_t pair = 0; pair < stage / 2; pair++)
45  {
46  for (std::size_t a = pair; a < len; a += stage)

48  std::size_t b = a + stage / 2;
49  std::complex<FloatT> t = field[a] - field[b];
